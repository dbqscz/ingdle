const champions = [ { name: "darlan", hint: "He is always number one!" }, { name: "vitoria", hint: "Your name is an achievement." }, { name: "rayane", hint: "She is very energetic." }, { name: "sarah", hint: "Your name is short and she used to have bangs." }, { name: "vivian", hint: "She owns a company." }, { name: "augusto", hint: "It's silly and bad-tempered." }, { name: "mavih", hint: "She is a good singer." }, { name: "aline", hint: "Wears braids and dances very well." }, ]; let remainingChampions = [...champions]; // Copia da lista original let currentChampion; let correctGuesses = 0; // Contador de acertos // Seleciona um campeão aleatório function selectRandomChampion() { if (remainingChampions.length === 0) { feedback.textContent = "No more members left!"; hint.textContent = ""; input.disabled = true; button.disabled = true; return null; } const randomIndex = Math.floor(Math.random() * remainingChampions.length); const champion = remainingChampions[randomIndex]; remainingChampions.splice(randomIndex, 1); // Remove da lista return champion; } // Inicializa o jogo function initGame() { if (correctGuesses >= 4) { feedback.textContent = "Game Over! You guessed 4 members correctly!"; feedback.style.color = "gold"; hint.textContent = ""; input.disabled = true; button.disabled = true; celebrationGif.style.display = "block"; // Exibe o GIF ao finalizar return; } currentChampion = selectRandomChampion(); if (currentChampion) { hint.textContent = `Hint: ${currentChampion.hint}`; feedback.textContent = ""; input.value = ""; input.disabled = false; button.disabled = false; celebrationGif.style.display = "none"; // Esconde o GIF ao iniciar } } // Elementos do DOM const input = document.getElementById("guess-input"); const button = document.getElementById("submit-btn"); const feedback = document.getElementById("feedback"); const hint = document.getElementById("hint"); const celebrationGif = document.getElementById("celebration-gif"); // Lógica do jogo button.addEventListener("click", () => { const userGuess = input.value.trim(); if (!userGuess) { feedback.textContent = "Write a name!"; feedback.style.color = "orange"; return; } if (userGuess.toLowerCase() === currentChampion.name.toLowerCase()) { correctGuesses++; feedback.textContent = `Correct! You have guessed ${correctGuesses}/4 members.`; feedback.style.color = "green"; celebrationGif.style.display = "block"; // Exibe o GIF setTimeout(initGame, 3000); // Troca de integrante após 3 segundos } else { feedback.textContent = "Try again :("; feedback.style.color = "red"; } input.value = ""; }); // Inicia o jogo ao carregar a página initGame();
